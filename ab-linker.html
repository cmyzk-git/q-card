function handleFile(file) { const reader = new FileReader(); reader.onload = function(evt) { const buffer = evt.target.result; let text = ""; try { // まず Shift_JIS として読む text = new TextDecoder('shift_jis').decode(buffer); // 文字化けが疑われる場合は UTF-8 で再読込 if (text.includes('�') || text.trim().length < 50) { text = new TextDecoder('utf-8').decode(buffer); } } catch (e) { try { text = new TextDecoder('utf-8').decode(buffer); } catch (e2) { alert('ファイルの読込に失敗しました'); return; } } loadedRawText = text; if (typeof processCurrentText === 'function') { processCurrentText(); } const dropText = document.getElementById('drop-text'); if (dropText) { dropText.innerHTML = `読み込み完了: ${file.name}`; } }; reader.onerror = function() { alert('ファイルの読込に失敗しました'); }; reader.readAsArrayBuffer(file); }
